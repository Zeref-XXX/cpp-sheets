class Solution {
public:
    string longestCommonPrefix(vector<string>& strs) {
        
        string result = "";
        for (int i = 0; i < strs[0].length(); i++) {
            char temp = strs[0][i];
            for (int j = 1; j < strs.size(); j++) {
                if (strs[j][i] != temp)
                    return result;
            }
            result += strs[0][i];
        }
        return result;
    }
};
