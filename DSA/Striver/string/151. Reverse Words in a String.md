class Solution {
public:
    string reverseWords(string s) {
        int i=0;
        int n=s.length();
        int l=0,r=0;
        reverse(s.begin(),s.end());
        while(i<n){ // i traverse the reversed string
            while(i<n && s[i]!=' '){
                // s[r]=s[i];i++;r++;
                s[r++]=s[i++];
            }
            if(l<r){
                reverse(s.begin()+l,s.begin()+r);
                s[r]=' '; r++;l=r;
            }
            i++;
        }
        return s.substr(0,r-1);
    }
};
