class Solution {
public:
    string largestOddNumber(string num) {
        int maxi = -1; 
        int l = 0;
        for (int i = num.length() - 1; i >= 0; i--) {
            int temp = num[i] - '0';
            if (temp % 2 != 0) {
                return num.substr(l, i + 1);
                
            }
        }
        return "";
    }
};
